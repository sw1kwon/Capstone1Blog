2. (이상치) 각 column별 이상치 확인

# 수치형 변수만 추출
numeric_cols = tidy.select_dtypes(include=['float64', 'int64']).columns

# 이상치 비율 저장용 리스트
outlier_info = []

# 각 수치형 column에 대해 이상치 비율 계산 (IQR 방식)
for col in numeric_cols:
    Q1 = tidy[col].quantile(0.25)
    Q3 = tidy[col].quantile(0.75)
    IQR = Q3 - Q1
    lower = Q1 - 1.5 * IQR
    upper = Q3 + 1.5 * IQR
    outliers = ((tidy[col] < lower) | (tidy[col] > upper)).sum()
    outlier_ratio = outliers / tidy.shape[0]
    outlier_info.append((col, outliers, outlier_ratio))

# 결과 DataFrame 정리
outlier_df = pd.DataFrame(outlier_info, columns=['Column', 'OutlierCount', 'OutlierRatio'])
outlier_df = outlier_df.sort_values(by='OutlierRatio', ascending=False)

        '비율(%)': ratios
    }).reset_index().rename(columns={'index': column})

    return summary

def calc_ratio_by_top_n(df, n):
    top_n = (
        df.dropna(subset=['판매지수', '분류'])
          .sort_values(['분류', '판매지수'], ascending=[True, False])
          .groupby('분류')
          .head(n)
          .copy()
    )

        top_n.groupby('분류')['추천평유무']
        .mean()
        .round(4)
        .reset_index(name=f'상위{n}추천평비율')
